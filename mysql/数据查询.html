# 数据查询

# 语法

SELECT    <列名> 
FROM      <表名> 
[WHERE    <查询条件表达式>] 
[ORDER BY <排序的列名>[ASC或DESC]]

# 查询结果排序 【ORDER BY】

-- 【desc】关键字 降序即结果按 id 降序排列
SELECT id,name,age FROM user order by id desc;

-- 【asc】关键字 升序即结果按 id 升序排列
SELECT id,name,age FROM user order by id asc;

-- 以逆字母顺序显示用户名，并以数字顺序显示年龄
SELECT id,name,age FROM user ORDER BY name DESC, age ASC;  -- 排序权重看先后

示例

查询成都的学生信息，并按 id 正序排列

SELECT id,name,idcard,age,city
FROM student
WHERE city = '成都'
ORDER BY id ASC
列名 表名 过滤条件 排序条件
SELECT id,name,idcard,age,city FROM student WHERE city= '成都' ORDER BY id asc

# 别名 【AS】

mysql> SELECT id,name,idcard,age AS 年龄, city AS home  
       FROM student 
       WHERE city = '成都' 
       ORDER BY age DESC;
+----+------+-------------------+------+------+
| id | name | idcard            | 年龄 | home |
+----+------+-------------------+------+------+
|  7 | 王五 | 51132********2154 |   20 | 成都 |
|  5 | 张三 | 51132********4568 |   15 | 成都 |
+----+------+-------------------+------+------+

# 查询空行

mysql> SELECT id,name,age,city
       FROM student
       WHERE city IS NULL OR city ='';
-- 结果
+----+------+------+------+
| id | name | age  | city |
+----+------+------+------+
|  8 | 赵六 |   11 |      |
+----+------+------+------+

|  5 | 张三 |   15 | 成都 | 中国    |
|  6 | 李四 |   50 | 绵阳 | 中国    |
|  7 | 王五 |   20 | 成都 | 中国    |
|  8 | 赵六 |   11 |      | 中国    |
+----+------+------+------+---------+

# 限制返回行数 【LIMIT】




 







mysql> SELECT id,name,age,city,'中国' as country
       FROM student
       ORDER BY age DESC
       LIMIT 2;
+----+------+------+------+---------+
| id | name | age  | city | country |
+----+------+------+------+---------+
|  6 | 李四 |   50 | 绵阳 | 中国    |
|  7 | 王五 |   20 | 成都 | 中国    |
+----+------+------+------+---------+

# 去除重复值 【DISTINCT】

例如查询所有同学来自哪些城市。

mysql> SELECT DISTINCT city FROM student;
-- 查询结果
+------+
| city |
+------+
| 成都 |
| 绵阳 |
|      |
+------+

# 运算符 【+】

在MYSQL中 + 号只能用作于数字;

-- 1、两个数都为数值型，进行加法的运算。
SELECT 1+1  -- 2

-- 2、+号左右两方有一个或者都为字符型，将会试图转换成数值型）如果转换成功，进行数值加法运算
SELECT 10 + '10'  -- 20

-- 3、如果转换不成功，则将字符型值转换为 0 进行加法运算。
SELECT 10 + 'hello'  -- 10

-- 4、只要一方为null，结果肯定为null。
SELECT 10 + NULL  -- NULL
