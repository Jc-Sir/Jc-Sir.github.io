# 数据表CRUD

# 查看表结构

-- DESC 表名
DESC user

# 创建数据表

CREATE TABLE test_table (
    `id` INT PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR ( 30 ) NOT NULL, 
    `remark` VARCHAR ( 255 ) DEFAULT NULL) 
    charset utf8;
    
-- 字段 id 为主键自增
-- 字段 name 为字符串类型 varchar 并不允许为 null
-- 字段 remark 为可为 null 字符串
-- 字符集为 utf8 ，如果不设置将继承数据库字符集

# 删除数据表

drop table IF EXISTS test_table 

drop table test_table,hdjs,copy_table_rename  // 多个表删除

-- 删除test_table  `IF EXISTS` 删除前判断是否存在

# 修改数据表【修改字段】

--增加身份证字段
ALTER TABLE table_name ADD COLUMN column_name datatype;
-- 单次增加多个字段
ALTER TABLE `student` 
ADD COLUMN `idcard` VARCHAR(15) UNIQUE NULL AFTER `id`,
ADD COLUMN name  VARCHAR( 16 ); 


-- 字段重命名
ALTER TABLE table_name RENAME COLUMN old_column_name TO new_column_name;  

-- 删除表字段
ALTER TABLE table_name DROP COLUMN column_name; 
-- 批量删除表字段
ALTER TABLE student 
DROP COLUMN gender,
DROP COLUMN idcard

-- 单次增加多个字段
ALTER TABLE student 
ADD COLUMN NAME VARCHAR ( 16 ) NOT NULL AFTER id,
ADD COLUMN idcard INT ( 18 ) UNIQUE,
ADD COLUMN gender INT ( 10 )

# 数据库约束【修改字段】

数据库约束是指对表中数据的限制和规范，以保证数据的有效性和完整性。常见的数据库约束包括：

# 主键约束（PRIMARY KEY）

用于唯一标识表中每一行数据，必须满足唯一性和非空性。

ALTER TABLE table_name ADD PRIMARY KEY (column_name);

# 外键约束（FOREIGN KEY）

用于建立表与表之间的关系，要求外键列必须在目标表中存在对应的主键值。

ALTER TABLE table_name ADD FOREIGN KEY (column_name) REFERENCES target_table (target_column);

# 唯一约束（UNIQUE）

用于保证某一列的取值在表中唯一，不允许重复。

ALTER TABLE table_name ADD UNIQUE (column_name);

# 非空约束（NOT NULL）

用于限制某一列的取值不为空。

ALTER TABLE table_name MODIFY column_name datatype NOT NULL;

# 检查约束（CHECK）

用于限制某一列的取值必须符合指定的条件或规则。

ALTER TABLE table_name ADD CHECK (column_name > 0);

# 数据查询

# 非条件查询

1、查询所有字段数据

select * from test_table

2、查询指定字段数据并排序字段

-- 查询description,cname 两个字段的数据
select description,cname from class;

# 条件查询

TIP

SELECT 列名称 FROM 表名称 WHERE 列 运算符 值

运算符 描述
= 等于
<> 不等于
> 大于
< 小于
>= 大于等于
<= 小于等于
between 在某范围内
like 某种搜索模式

注释：在某些版本的 SQL 中，操作符 <> 可以写为 !=。

1、查找用户名为admin的数据

select * from user where username = 'admin'

2、查询包含/不包含关键词的数据 【like】

-- 查询包含关键字数据
select * from user where username like '%adm%'

-- 查询不包含关键字数据
select * from user where username not like '%adm%'

3、合并列返回查询结果 【concat】

SELECT CONCAT(firstname," ",lastname) AS fullname  from user

4、指定多条件查询 【and】

-- 选取id>4 且 gender = 1 的数据
SELECT * from user WHERE id > 4 AND gender = 1

5、或 条件查询 【OR】

-- 查询id>4 或者status 值为0 的数据
SELECT * from user WHERE id > 4 OR STATUS = 0

6、去除重复值 【DISTINCT】

-- SELECT DISTINCT 列名称 FROM 表名称
-- 查找用户有几种状态
SELECT DISTINCT status from user

7、查找数据范围内/外数据 【BETWEEN AND】

-- 查找用户年龄在20-30的数据
select * from user where age BETWEEN 20 AND 30;

-- 查找用户年龄不在20-30的数据
select * from user where age NOT BETWEEN 20 AND 30;

8、查找在某个类别下的数据 【in】

-- 查找状态0的数据
SELECT * FROM user where status IN(0);
-- 查找状态 0,1 的数据
SELECT * FROM user where status IN(0,1);

-- 查找状态不是 0,1 的数据
SELECT * FROM user where status NOT IN(0,1);

-- 查询班级1、2的学生数据
SELECT * FROM user where status IN(1,2);

9、NULL值

-- 查询firstname 为null 的数据
SELECT * FROM user where firstname is null;

-- 查询firstname 不为null 的数据
SELECT * FROM user where firstname is not null;

-- 查询结果中对没状态的数据显示未知状态
select username,if(status is null,'未知状态',firstname) from user;
# 也可以使用IFNULl 函数简化操作
select username,ifnull (status ,'未知状态') from user;

# 查询结果排序 【order by】

-- 【desc】关键字 降序即结果按年龄降序排列
SELECT username,age FROM user order by age desc;

-- 【asc】关键字 升序即结果按年龄升序排列
SELECT username,age FROM user order by age asc;

-- 以逆字母顺序显示用户名，并以数字顺序显示年龄
-- 排序权重看先后
SELECT username, age FROM user ORDER BY username DESC, age ASC

# 复制表及数据

1、根据已经存在的表结构创建新表

create table test_table_copy like test_table;

2、复制其他表的数据

insert into test_table_copy select * from test_table;

3、复制指定字段数据

insert into test_table_copy (id,name) select id,name from test_table;

4、复制表时同时复制数据

create table test_table_copy select * from test_table;

5、复制指定字段，并为不同名字段起别名

CREATE TABLE copy_table_rename (
	id INT PRIMARY KEY AUTO_INCREMENT,
    cname VARCHAR ( 30 )) 
SELECT id, NAME AS cname FROM test_table

# 添加数据【插入数据行】

-- 语法
INSERT INTO 表名 (列名1,列名2,...) VALUES (值1,值2,...);

# 添加一条数据

insert into test_table set name="admin",remark="添加一条数据测试",id=5;

INSERT INTO test_table (name) VALUES('mysql');
-- 因为 remark 设置为 null 所以第二个记录没有设置值时使用默认的 null 值。

# 添加多条数据

INSERT INTO test_table (name,remark) 
VALUES
('zhangsan','学习Mysql'),
('lisi',"mysql添加数据");

注意事项

  • 每次插入一行数据，不能只插入一部分数据，插入的数据是否有效将按照整行的完整性要求来检验
  • 每个数据值的数据类型、精度、位数必须与要应的列名精确匹配
  • 不能为标识符指定值
  • 如果某字段设置为不能为空，则必须插入数据
  • 插入数据时还要符合检查性约束的要求
  • 有缺省值的列，可以使用DEFAULT关键字来代替插入实际的值

# 修改数据【更新数据行】

# 语法

UPDATE 表名称 SET 列名称 = 新值 WHERE 列名称 = 某值

# 示例

-- 将user表中 username为Fu Yunxi 的值改为 admin
UPDATE `user` set username = 'admin' WHERE username = 'Fu Yunxi'

-- 修改一行中的若干列
UPDATE `user` set username = 'admin',password = 123456 WHERE username = 'Fu Yunxi'

WARNING

  • 多列时用逗号隔开，一定要加更新条件以免错误更新
  • 多个联合条件使用 AND id=7 AND idcard='410787'
  • 判断某字段是否为空 email is null or email = ''

# 删除数据【删除数据行】

# 删除某一行

-- 语法
DELETE FROM 表名称 WHERE 列名称 = 值

-- 示例： 删除年龄小于18岁的记录
DELETE FROM user WHERE age < 18;

WARNING

  • Delete语句是对整行进行操作，因此不需要提供列名
  • 如果要删除主表数据，则要先删除子表记录

# 删除所有行

-- 删除user表中所有数据
delete from user

# 截断表 【TRUNCATE】

-- 语法
TRUNCATE  TABLE  表名

-- 示例：截断学生表
TRUNCATE TABLE student

DANGER

数据全部清空,但表结构、列、约束等不被改动 不能用于有外键约束引用的表 标识列重新开始编号 因为要删除的数据不会写入日志， 数据也不能恢复，所以工作中请尽量不要使用此命令
