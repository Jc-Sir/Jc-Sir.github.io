# 字符串函数

# LENGTH

WARNING

  • 计算单位：字节
  • utf8编码：一个汉字三个字节，一个数字或字母一个字节。
  • gbk编码：一个汉字两个字节，一个数字或字母一个字节。
SELECT
	id,
	name,
	LENGTH( name ) AS NameLenth 
FROM
	student;
	
-- 查询结果
+----+-------+-----------+
| id | name  | NameLenth |
+----+-------+-----------+
|  1 | admin |         5 |
|  5 | 张三  |         6 |
|  6 | 李四  |         6 |

# CHAR_LENGTH

WARNING

  • 计算单位：字符
  • 不管汉字还是数字或者是字母都算是一个字符
SELECT
	id,
	name,
	CHAR_LENGTH( name ) AS NameLenth 
FROM
	student;
	
-- 查询结果
+----+-------+-----------+
| id | name  | NameLenth |
+----+-------+-----------+
|  1 | admin |         5 |
|  5 | 张三  |         2 |
|  6 | 李四  |         2 |
+----+-------+-----------+

-- 检验是否存在中午字符  char_lenght(str) <> length(str)
SELECT 
  name,
	length( name ),
	char_length( name ) 
FROM
	student 
WHERE
	length( name )<> char_length( name );

-- 查询结果
+------+--------------+-------------------+
| name | length(name) | char_length(name) |
+------+--------------+-------------------+
| 张三 |            6 |                 2 |
| 李四 |            6 |                 2 |
...
+------+--------------+-------------------+

# 截取字符串【SUBSTR】

TIP

SUBSTR(str, start, length)

  • str：源字符串 【必须】
  • start：起始位置。 可以是正数或负数。 如果是正数，则此函数从字符串的开头提取。 如果是负数，此函数从字符串的末尾提取。
  • length：要提取的字符长度。 如果省略，将返回整个字符串（从 start 位置开始）

WARNING

注意：mysql中的 start是从1开始的。

-- 截取从指定索引处开始的所有的字符【包含指定索引处的字符】
SELECT SUBSTR('hello',2);
+-------------------+
| SUBSTR('hello',2) |
+-------------------+
| ello              |
+-------------------+

-- 截取从指定索引处开始的指定数量的字符
SELECT SUBSTR('hello',2,3);
+---------------------+
| SUBSTR('hello',2,3) |
+---------------------+
| ell                 |
+---------------------+

-- start 为负
SELECT SUBSTR('hello',-5,2); -- he

# 去掉空格【TRIM】

TIP

TRIM([{BOTH | LEADING | TRAILING} [要删除得字符串] FROM] 源字符串)

  • 若分类符BOTH、LEADIN或TRAILING中没有一个是给定的,则假设为BOTH。
  • 要删除得字符串为可选项，在未指定情况下，可删除空格。
  • 只能是针对于前后字符 源字符中间不受影响。
SELECT TRIM('  hello  ');  -- 删除前后空格
SELECT LTRIM('  hello  '); -- 去左侧空格
+--------------------+
| LTRIM('  hello  ') |
+--------------------+
| hello              |
+--------------------+

SELECT RTRIM('  hello  '); -- 去右侧空格
+--------------------+
| RTRIM('  hello  ') |
+--------------------+
|   hello            |
+--------------------+

SELECT TRIM('x' FROM 'xxhelloxxx'); -- 删除左右x字符
SELECT TRIM(BOTH 'x' FROM 'xxhelloxxx'); -- 删除左右x字符
+-----------------------------+
| TRIM('x' FROM 'xxhelloxxx') |
+-----------------------------+
| hello                       |
+-----------------------------+

SELECT TRIM(LEADING 'x' FROM 'xxhelloxxx'); -- 删除开头的x字符
+-------------------------------------+
| TRIM(leading 'x' FROM 'xxhelloxxx') |
+-------------------------------------+
| helloxxx                            |
+-------------------------------------+

 SELECT TRIM(TRAILING 'x' FROM 'xxhelxloxxx'); -- 删除末尾的 x 字符
+---------------------------------------+
| TRIM(TRAILING 'x' FROM 'xxhelxloxxx') |
+---------------------------------------+
| xxhelxlo                              |
+---------------------------------------+

# 字符串左右补齐 【LPAD/RPAD】

SELECT LPAD(str, len, padstr) -- padstr可选
/**
1、如果 len 小于原字符串 str 的长度， str 将被截断到 len 的长度。
2、如果 len 为负数时，LPAD() 函数将返回 NULL。
3、当任意一个参数为 NULL 时， LPAD() 函数将返回 NULL
 */

SELECT LPAD('1',3,'@'); -- 左补齐 不足位左侧以 @ 补齐
+-----------------+
| LPAD('1',3,'@') |
+-----------------+
| @@1             |
+-----------------+
-- 常见示例
SELECT
    LPAD('oh', 10, 'o'),
    LPAD('oh', 1, 'o'),
    LPAD('oh', -1, 'o'),
    LPAD('World', 15, 'Hello');

+---------------------+--------------------+---------------------+----------------------------+
| LPAD('oh', 10, 'o') | LPAD('oh', 1, 'o') | LPAD('oh', -1, 'o') | LPAD('World', 15, 'Hello') |
+---------------------+--------------------+---------------------+----------------------------+
| oooooooooh          | o                  | NULL                | HelloHelloWorld            |
+---------------------+--------------------+---------------------+----------------------------+

-- 语法
RPAD(str, len, padstr)
/**
- 如果 len 小于原字符串 str 的长度， str 将被截断到 len 的长度。
- 如果 len 为负数时，RPAD() 函数将返回 NULL。
- 当任意一个参数为 NULL 时， RPAD() 函数将返回 NULL。
*/

SELECT
    RPAD('oh', 10, 'h'),
    RPAD('oh', 1, 'h'),
    RPAD('oh', -1, 'h'),
    RPAD('Hello', 13, 'World'),
    RPAD('Hello', 13, NULL);

+---------------------+--------------------+---------------------+----------------------------+-------------------------+
| RPAD('oh', 10, 'h') | RPAD('oh', 1, 'h') | RPAD('oh', -1, 'h') | RPAD('Hello', 13, 'World') | RPAD('Hello', 13, NULL) |
+---------------------+--------------------+---------------------+----------------------------+-------------------------+
| ohhhhhhhhh          | o                  | NULL                | HelloWorldWor              | NULL
|
+---------------------+--------------------+---------------------+----------------------------+-------------------------+

# 数字格式化【FORMAT】

# 语法

FORMAT(number, decimal_places)

  • number: 需要格式化的数字。 【必需】
  • decimal_places：要保留的小数位数。【必需】

# 返回值

FORMAT() 函数返回具有指定小数位数的类似于 #,###,###.## 格式的字符串。

  • 如果参数 number 为字符或字符串，FORMAT() 将按照如下规则尝试转为数字：
  • 如果以数字开头，则将开头的数字部分转为数字。
  • 如果不能转为数字，则按照 0 处理。
  • 如果参数 number 为 NULL，FORMAT() 函数将会返回 NULL。
  • 如果参数 decimal_places 为 0，FORMAT() 函数返回的内容没有小数部分。
SELECT
    FORMAT(111111, 2),
    FORMAT(111111.111, 2),
    FORMAT(111111.111, 0),
    FORMAT('111111', 2),
    FORMAT('11111A', 2),
    FORMAT('A11111', 2),
    FORMAT(NULL, 2)\G

*************************** 1. row ***************************
    FORMAT(111111, 2): 111,111.00
FORMAT(111111.111, 2): 111,111.11
FORMAT(111111.111, 0): 111,111
  FORMAT('111111', 2): 111,111.00
  FORMAT('11111A', 2): 11,111.00
  FORMAT('A11111', 2): 0.00
      FORMAT(NULL, 2): NULL

# LEFT/RIGHT

TIP

LEFT() 函数从指定字符串的左侧返回指定数量的字符组成的字符串。 RIGHT() 函数从指定字符串的右侧返回指定数量的字符组成的字符串。

# 语法

TIP

LEFT | RIGHT (string, length)

  • string 必需的。需要从中提取字符的字符串。
  • length 必需的。需要从字符串中提取字符的数量。

# 返回值

  • 如果 length 超出了 string 的长度，LEFT() / RIGHT()函数返回 string。
  • 如果 length 为零或负数，LEFT() / RIGHT()函数将返回一个空串。
  • 当任意一个参数为 NULL 时， LEFT() / RIGHT() 函数将返回 NULL。

-- 常见示例
SELECT
    LEFT('Hello', 1),
    LEFT('Hello', 2),
    LEFT('Hello', 3),
    LEFT('Hello', 0),
    LEFT('Hello', -1),
    LEFT('Hello', NULL),
    LEFT(NULL, NULL)\G
*************************** 1. row ***************************
   LEFT('Hello', 1): H
   LEFT('Hello', 2): He
   LEFT('Hello', 3): Hel
   LEFT('Hello', 0):
  LEFT('Hello', -1):
LEFT('Hello', NULL): NULL
   LEFT(NULL, NULL): NULL


SELECT
    RIGHT('Hello', 1),
    RIGHT('Hello', 2),
    RIGHT('Hello', 3),
    RIGHT('Hello', 0),
    RIGHT('Hello', -1),
    RIGHT('Hello', NULL),
    RIGHT(NULL, NULL)\G
*************************** 1. row ***************************
   RIGHT('Hello', 1): o
   RIGHT('Hello', 2): lo
   RIGHT('Hello', 3): llo
   RIGHT('Hello', 0):
  RIGHT('Hello', -1):
RIGHT('Hello', NULL): NULL
   RIGHT(NULL, NULL): NULL

# 四舍五入 【ROUND】

WARNING

  • 如果 d 大于等于 x 的小数位数，则返回原数字。
  • 如果 d 小于 x 的小数位数，则将 x 的小数位四舍五入到 d 位后返回。
  • 如果 d 为负数，ROUND() 函数将会从小数点开始向整数部分的 d 位数字替换为 0。
  • 如果任意一个参数为 NULL，ROUND() 函数将会返回 NULL。

# 语法

ROUND(x, d)
-- 参数 x :被处理的数字。
-- 参数 d :需要保留的小数位数。

# 示例

SELECT
    ROUND(123.179, 1),
    ROUND(123.179, 2),
    ROUND(123.179, 4),
    ROUND(123.179, 0),
    ROUND(123.179, -1),
    ROUND(123.179, -2),
    ROUND(123.179, NULL)\G

*************************** 1. row **************************
   ROUND(123.179, 1): 123.2
   ROUND(123.179, 2): 123.18
   ROUND(123.179, 4): 123.179
   ROUND(123.179, 0): 123
  ROUND(123.179, -1): 120
  ROUND(123.179, -2): 100
ROUND(123.179, NULL): NULL

# 当前日期时间 【NOW】

MySQL NOW() 函数按 YYYY-MM-DD hh:mm:ss 格式返回当前时间和日期。

-- 返回系统的当前时间。
SELECT NOW(), NOW() + 1;

+---------------------+----------------+
| NOW()              | Now() + 1        |
+---------------------+----------------+
| 2023-05-04 22:42:52 | 20230504224253 |
+---------------------+----------------+

TIP

注意： NOW() + 0 的结果为 YYYYMMDDhhmmss 格式。NOW() + N意味着当前时间加上 N 秒。

WARNING

与 SYSDATE()函数不同，NOW() 函数返回的是所在的语句、函数、存储过程开始执行的时间，并不一定是执行的 NOW() 语句时确切的系统时间。

SELECT
    NOW(),
    SYSDATE(),
    SLEEP(2),
    NOW(),
    SYSDATE()\G
*************************** 1. row ***************************
    NOW(): 2023-05-04 22:44:49
SYSDATE(): 2023-05-04 22:44:49
 SLEEP(2): 0
    NOW(): 2023-05-04 22:44:49
SYSDATE(): 2023-05-04 22:44:51

# 字符串转日期格式【STR_TO_DATE】

STR_TO_DATE() 函数将指定的字符串根据指定日期格式转为日期/时间。 STR_TO_DATE() 函数与 DATE_FORMAT() 函数是相反的。

-- 语法
STR_TO_DATE(str, format)

-- 参数
-- str ：必需的。需要格式化成日期的字符串。
-- format ：必需的。格式化字符串。
序号 格式符 功能
1 %Y 4位的年份
2 %y 2位的年份
3 %m 月份(01,02)
4 %c 月份(1,2)
5 %d 日(01,02)
6 %H 小时(24小时制)
7 %h 小时(12小时制)
8 %i 分钟(00,01)
9 %s 秒(00,01)
-- 转为日期
 SELECT STR_TO_DATE('2023/05/04', '%Y/%m/%d');
+---------------------------------------+
| STR_TO_DATE('2023/05/04', '%Y/%m/%d') |
+---------------------------------------+
| 2023-05-04                            |
+---------------------------------------+

-- 转为日期时间
SELECT STR_TO_DATE('2023/05/04 23:06:10', '%Y/%m/%d %H:%i:%S');
+---------------------------------------------------------+
| STR_TO_DATE('2023/05/04 23:06:10', '%Y/%m/%d %H:%i:%S') |
+---------------------------------------------------------+
| 2023-05-04 23:06:10                                     |
+---------------------------------------------------------+
1 row in set (0.00 sec)


select STR_TO_DATE('12-30 1984','%m-%d %Y');
+--------------------------------------+
| STR_TO_DATE('12-30 1984','%m-%d %Y') |
+--------------------------------------+
| 1984-12-30                           |
+--------------------------------------+

# 加上指定时间间隔【DATE_ADD】

DATE_ADD() 函数在指定的日期/时间上加上指定到时间间隔加并返回新的日期/时间。

# 语法

DATE_ADD(date, INTERVAL value unit)  -- INTERVAL:间隔

-- 参数
date 必需的。需要操作的日期。
value 必需的。时间/日期间隔。正数和负数都是允许的。
unit 必需的。时间/日期间隔的单位。

-- 返回值
1、如果 date 参数是 DATE 类型，并且时间间隔为 YEAR, MONTH 或者 DAY，返回 DATE。
2、如果 date 参数是 DATE 类型，并且时间间隔为 HOURS, MINUTES 或者 SECONDS，返回 DATETIME。
3、如果 date 参数是 DATETIME 类型，返回 DATETIME。
4、如果 date 参数是 TIME 类型，并且时间间隔为 YEAR, MONTH 或者 DAY，返回 DATETIME。
5、如果 date 参数是 TIME 类型，并且计算只涉及 HOURS, MINUTES 和 SECONDS 部分，返回 TIME。（开始于 MySQL 8.0.28）

间隔单位

  • MICROSECOND
  • SECOND
  • MINUTE
  • HOUR
  • DAY
  • WEEK
  • MONTH
  • QUARTER
  • YEAR
  • SECOND_MICROSECOND
  • MINUTE_MICROSECOND
  • MINUTE_SECOND
  • HOUR_MICROSECOND
  • HOUR_SECOND
  • HOUR_MINUTE
  • DAY_MICROSECOND

# 两个日期值之间的天数【DATEDIFF】

DATEDIFF(date1, date2)
-- 参数：日期或者日期时间表达式

-- 返回值
1、date1 > date2 正数  -- 如果 date1 的日期在 date2 的日期之后。
2、date1 = date2 0 
3、date1 < date2 负数
4、参数不合法     NULL

# 示例

SELECT
    DATEDIFF('2022-02-28', '2022-02-28'),
    DATEDIFF('2022-02-28', '2022-02-28 10:10:10'),
    DATEDIFF('2022-02-28 10:10:10', '2022-02-28'),
    DATEDIFF('2022-02-28', '2022-02-27'),
    DATEDIFF('2022-02-28', '2022-02-27 10:10:10'),
    DATEDIFF('2022-02-28 10:10:10', '2022-02-27'),
    DATEDIFF('2022-02-30', '2022-02-28 10:10:10'),
    DATEDIFF('Not A DATEDIFF', 'Not A DATEDIFF'),
    DATEDIFF(NULL, '2022-02-28')\G

*************************** 1. row ***************************
         DATEDIFF('2022-02-28', '2022-02-28'): 0
DATEDIFF('2022-02-28', '2022-02-28 10:10:10'): 0
DATEDIFF('2022-02-28 10:10:10', '2022-02-28'): 0
         DATEDIFF('2022-02-28', '2022-02-27'): 1
DATEDIFF('2022-02-28', '2022-02-27 10:10:10'): 1
DATEDIFF('2022-02-28 10:10:10', '2022-02-27'): 1
DATEDIFF('2022-02-30', '2022-02-28 10:10:10'): NULL  -- 2月没有30天
 DATEDIFF('Not A DATEDIFF', 'Not A DATEDIFF'): NULL
                 DATEDIFF(NULL, '2022-02-28'): NULL
