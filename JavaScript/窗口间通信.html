// worker.js
var portList = [];

onconnect = function (e) {
  var port = e.ports[0];
  ensurePorts(port);
  port.onmessage = function (e) {
    var data = e.data;
    disptach(port, data);
  };
  port.start();
};

function ensurePorts(port) {
  if (portList.indexOf(port) < 0) {
    portList.push(port);
  }
}

function disptach(selfPort, data) {
  portList
    .filter((port) => selfPort !== port)
    .forEach((port) => port.postMessage(data));
}
